```javascript
//함정이 많은 문제
//연산자의 사용( || ) && 단항 더하기 연산자 및 Number메소드를 사용할 때의 형변환 주의
function solution(s) {
    const isInt = +(s) === parseInt(s, 10) 
    return (isInt) && ( s.length.toString() === '4' || s.length.toString() === '6')   ? true : false 
}
```
